\section{Foundations of Widget Theory}

\begin{defbox}[Definition: Widget Norm]
$\|w\|_W = \sup_{k \ge 1}|w_k| \cdot 2^{-k}$. Finite if $\|w\|_W < \infty$.
\end{defbox}

\begin{thmbox}[Theorem: Widget Approximation]
Bounded sequence in $W$ has a norm-convergent subsequence.
\end{thmbox}

\begin{proof}[Proof sketch]
Apply BW to each component, extracting nested subsequences. Diagonalize.
Tail bound $\sup_{k>K}|w_k|2^{-k} \le M$ controls the remainder.
\end{proof}

\begin{exbox}[Example]
$w^{(n)}_k = 1/(n+k)$: bounded by $M=1$, converges to $w^*=0$.
\end{exbox}

\begin{thmbox}[Lemma: Tail Bound]
$\|w\|_W \le M \Rightarrow \sup_{k>K}|w_k| \cdot 2^{-k} \le M$.
\end{thmbox}

\begin{proof}[Proof sketch]
Sup over a subset $\le$ sup over the full set.
\end{proof}
