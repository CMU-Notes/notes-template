\section*{CH 1: Foundations}

\subsection{Widget Norm}
$\|w\|_W = \sup_{k \ge 1} |w_k| \cdot 2^{-k}$. Finite if $< \infty$.

\subsection{Triangle Inequality}
$\|u+v\|_W \le \|u\|_W + \|v\|_W$.\\
\pf{Bound each $|u_k+v_k|2^{-k}$ by norms, take sup.}

\subsection{Widget Approximation Thm}
$(w^{(n)})$ bounded $\Rightarrow$ $\exists$ convergent subsequence.\\
\pf{BW per component + diagonal + tail bound.}

\subsection{Tail Bound}
$\|w\|_W \le M \Rightarrow \sup_{k>K}|w_k|2^{-k} \le M$.\\
\pf{Sup over subset $\le$ sup over full set.}
